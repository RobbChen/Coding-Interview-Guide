## 计算数组的小和

**题目：**
>给定一个无序单链表的头节点head,删除其中值重复出现的节点。
```
例如：l->2->3->3->4->4->2->1->null，删除值重复的节点之后为l->2->3->4->null。
请按以下要求实现两种方法。
方法l：如果链表长度为N，时间复杂度达到O(N)。
方法2：额外空间复杂度为O(1)。
```

```java
public class Problem_13_SmallSum {

	public static int getSmallSum(int[] arr) {
		if (arr == null || arr.length == 0) {
			return 0;
		}
		return func(arr, 0, arr.length - 1);
	}

	public static int func(int[] s, int l, int r) {
		if (l == r) {
			return 0;
		}
		int mid = (l + r) / 2;
		return func(s, l, mid) + func(s, mid + 1, r) + merge(s, l, mid, r);
	}

	public static int merge(int[] s, int left, int mid, int right) {
		int[] h = new int[right - left + 1];
		int hi = 0;
		int i = left;
		int j = mid + 1;
		int smallSum = 0;
		while (i <= mid && j <= right) {
			if (s[i] <= s[j]) {
				smallSum += s[i] * (right - j + 1);
				h[hi++] = s[i++];
			} else {
				h[hi++] = s[j++];
			}
		}
		for (; (j < right + 1) || (i < mid + 1); j++, i++) {
			h[hi++] = i > mid ? s[j] : s[i];
		}
		for (int k = 0; k != h.length; k++) {
			s[left++] = h[k];
		}
		return smallSum;
	}

	public static void swap(int[] arr, int index1, int index2) {
		int tmp = arr[index1];
		arr[index1] = arr[index2];
		arr[index2] = tmp;
	}

	public static void main(String[] args) {
		int[] arr = { 3, 1, 2, 4, 6, 2, 7, 8, 1 };
		System.out.println(getSmallSum(arr));

	}

}
```
